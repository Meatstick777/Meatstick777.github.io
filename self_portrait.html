This is my Self-Portrait which I made using the Processing application.

Here is the code:

function setup() {
  createCanvas(600, 600);
}

function draw() {
  background(102, 255, 204);

  // Face
  fill(210, 180, 140);
  ellipse(300, 300, 210, 245);

  // Ears
  fill(210, 180, 140);
  ellipse(195, 315, 29, 44);
  ellipse(405, 315, 29, 44);

  // Left eye
  fill(255);
  circle(260, 300, 30);
  fill(0);
  circle(260, 300, 19);

  // Right eye
  fill(255);
  circle(340, 300, 30);
  fill(0);
  circle(340, 300, 19);

  // Eyelids
  fill(210, 180, 140);
  arc(260, 300, 30, 30, PI, TWO_PI);
  arc(340, 300, 30, 30, PI, TWO_PI);

  // Smile with teeth
  noStroke();
  fill(255);
  arc(300, 360, 120, 90, 0, PI);

  // Teeth block
  fill(300);
  rect(250, 360, 100, 35, 30);

  // Tooth separator lines
  stroke(0);
  strokeWeight(2);
  let startX = 250;
  let endX = 350;
  let numLines = 5;
  for (let i = 1; i < numLines; i++) {
    let x = startX + (i * (endX - startX) / numLines);
    line(x, 360, x, 395);
  }

  noStroke();

  // Hair as a half-circle
  fill(0);
  arc(300, 295, 210, 260, PI, 0);

  // Neck
  fill(210, 180, 140);
  rect(285, 420, 30, 50, 5);

  // Body/torso
  fill(100);
  beginShape();
  vertex(250, 470);
  vertex(350, 470);
  vertex(330, 600);
  vertex(270, 600);
  endShape(CLOSE);

  // Arms
  fill(100);
  beginShape();
  vertex(250, 470);
  vertex(220, 480);
  vertex(230, 620);
  vertex(250, 600);
  endShape(CLOSE);

  beginShape();
  vertex(350, 470);
  vertex(380, 480);
  vertex(370, 620);
  vertex(350, 600);
  endShape(CLOSE);
}
