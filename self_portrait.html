This is my Self-Portrait which I made using the Processing application.

function setup() {
  createCanvas(600, 600);
}

function draw() {
  background(102, 255, 204);

  // Face
  fill(210, 180, 140);
  ellipse(300, 300, 210, 245);

  // Ears
  fill(210, 180, 140);
  ellipse(195, 315, 29, 44);
  ellipse(405, 315, 29, 44);

  // Left eye
  fill(255);
  circle(260, 300, 30);
  fill(0);
  circle(260, 300, 19);

  // Right eye
  fill(255);
  circle(340, 300, 30);
  fill(0);
  circle(340, 300, 19);
}
